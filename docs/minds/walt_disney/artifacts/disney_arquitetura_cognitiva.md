    quality_score = avaliar_disney_quality(situação)
    if quality_score < 95:
        return "REFAZER_ATÉ_PERFEITO"
    
    # Check de Inovação
    if já_foi_feito_antes(situação):
        return "PLUS_ATÉ_REVOLUCIONAR"
    
    # Validação de Controle
    if não_tenho_controle_total(situação):
        return "ADQUIRIR_CONTROLE_OU_REJEITAR"
    
    # Teste de Magia
    if not cria_momento_mágico(situação):
        return "REFORMULAR_COMPLETAMENTE"
    
    # Viabilidade Roy
    if roy_diz_impossível(situação):
        return "CONVENCER_MANIPULAR_PROCEDER"
    
    # Execução
    return "OBSESSÃO_TOTAL_ATÉ_PERFEIÇÃO"
