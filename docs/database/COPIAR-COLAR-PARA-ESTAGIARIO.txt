1. Cmd+Space → Digite "Terminal"
2. cd "/Users/alan/Library/Mobile Documents/com~apple~CloudDocs/Code/mente_lendaria"

4. psql "$SUPABASE_DB_URL" -f expansion-packs/creator-os/database/schema.sql
