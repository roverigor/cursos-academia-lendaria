Desfazendo Erros
	Cenarios Comuns
		Mudancas para descartar
		Editar commit/mensagem
		Commit branch errada
		Guardar temporariamente
		Pushed algo errado
	git restore
		Desfazer nao commitadas
		Arquivo especifico
			git restore index.js
			git checkout -- index.js
		Todos arquivos
			git restore .
		ATENCAO
			Mudancas perdidas permanentemente
		Unstage
			git restore --staged
			git reset HEAD
	git reset
		Desfazer commits
		CUIDADO
			Reescreve historico
			Nao usar em pushed compartilhado
		3 Modos
			--soft
				Mantém: Working + Staging
				Volta: Repository
			--mixed (padrao)
				Mantém: Working
				Limpa: Staging
				Volta: Repository
			--hard (PERIGOSO)
				Limpa: Working + Staging
				Volta: Repository
		soft reset
			git reset --soft HEAD~1
			Util: editar mensagem
		mixed reset
			git reset HEAD~1
			Util: reorganizar commits
		hard reset
			git reset --hard HEAD~1
			PERIGO: perde mudancas
		Voltar multiplos
			HEAD~3 (3 commits)
			abc1234 (commit especifico)
		Ver historico
			git reflog
			Recuperar apos reset
	git revert
		Desfazer publicamente
		Ja fez push
			Nao pode reset
		Cria novo commit
			Desfaz mudancas
			Preserva historico
		Comando
			git revert HEAD
			git revert abc1234
		Diferenca reset
			Reset: reescreve
			Revert: adiciona commit
		Sem commit auto
			git revert --no-commit
			Revisar antes commitar
	git stash
		Guardar temporariamente
		Quando
			Mudar branch urgente
			Trabalho incompleto
		Stash basico
			git stash
			git stash save "mensagem"
		Listar
			git stash list
		Recuperar
			git stash pop (remove)
			git stash apply (mantem)
		Ver conteudo
			git stash show
			git stash show -p
		Deletar
			git stash drop
			git stash clear (todos)
		Incluir untracked
			git stash -u
	git commit --amend
		Corrigir ultimo commit
		Editar mensagem
			git commit --amend
			git commit --amend -m "nova"
		Adicionar arquivo
			git add forgotten.js
			git commit --amend --no-edit
		IMPORTANTE
			Reescreve commit
			Nao se ja pushed
	git clean
		Remover nao rastreados
		Dry-run
			git clean -n
		Deletar
			git clean -f (arquivos)
			git clean -fd (+ diretorios)
			git clean -fdx (+ ignorados)
		CUIDADO
			Deleta permanentemente
	Arvore Decisao
		Nao commitou
			Descartar → restore
			Guardar → stash
			Limpar → clean
		Commitou localmente
			Nao pushed
				Editar ultimo → amend
				Desfazer mantendo → reset --soft
				Desfazer descartando → reset --hard
			Ja pushed
				Desfazer → revert (nunca reset!)
		Branch errada
			Nao pushed → reset + cherry-pick
			Pushed → PR da branch errada
	Recuperando Desastres
		Commit apos reset
			git reflog
			Encontrar commit perdido
			git reset --hard abc1234
		Arquivo deletado
			git log --diff-filter=D
			git checkout abc1234^ -- file.js
	Boas Praticas
		Faca
			stash para mudar contexto
			revert em pushed
			Testar apos desfazer
			reflog para recuperar
		Evite
			reset --hard sem backup
			reset em pushed
			push --force compartilhado
			clean sem -n antes
	Comandos Emergencia
		"Estraguei tudo"
			git reflog
			git reset --hard HEAD@{X}
		"Branch errada"
			git reset --hard HEAD~1
			git checkout branch-certa
			git cherry-pick abc1234
		"Voltar para GitHub"
			git fetch origin
			git reset --hard origin/main
