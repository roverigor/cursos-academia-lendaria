```javascript
const places = $input.all()[0].json.data;

return places.map(place => ({
  json: {
    name: place.name,
    phone: place.phone || null,
    email: place.emails?.[0] || null,
    website: place.site,
    address: place.full_address,
    city: place.city,
    rating: place.rating,
    reviews: place.reviews,
    category: place.category,
    source: 'google_maps',
    source_id: place.place_id
  }
}));
```
