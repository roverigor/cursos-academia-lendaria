### [INSTALAÇÃO VIA PORTAINER] - 2 minutos

**[TELA: Portainer]**

> Stacks → Add Stack → Nome: `calcom`
>
> **[COLA O COMPOSE]**

```yaml
version: "3.8"
services:
  calcom:
    image: calcom/cal.com:latest
    container_name: calcom
    restart: unless-stopped
    environment:
      - DATABASE_URL=postgresql://calcom:senha-do-banco@calcom-db:5432/calcom
      - NEXTAUTH_SECRET=sua-chave-secreta-muito-longa-aqui
      - CALENDSO_ENCRYPTION_KEY=outra-chave-secreta-32-chars!!!
      - NEXT_PUBLIC_WEBAPP_URL=https://agenda.seudominio.com.br
      - NEXT_PUBLIC_LICENSE_CONSENT=agree
    labels:
      - "traefik.enable=true"
      - "traefik.http.routers.calcom.rule=Host(`agenda.seudominio.com.br`)"
      - "traefik.http.routers.calcom.entrypoints=websecure"
      - "traefik.http.routers.calcom.tls.certresolver=letsencrypt"
      - "traefik.http.services.calcom.loadbalancer.server.port=3000"
    networks:
      - traefik_proxy
      - calcom_internal

  calcom-db:
    image: postgres:15
    container_name: calcom-db
    restart: unless-stopped
    environment:
      - POSTGRES_USER=calcom
      - POSTGRES_PASSWORD=senha-do-banco
      - POSTGRES_DB=calcom
    volumes:
      - calcom_db:/var/lib/postgresql/data
    networks:
      - calcom_internal

volumes:
  calcom_db:

networks:
  traefik_proxy:
    external: true
  calcom_internal:
```

> **[EXPLICA]**
>
> Troca `seudominio.com.br`, senhas e chaves.
>
> **[DEPLOY]**
